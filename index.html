// ---- View counter with countdown to milestone ---- (function () { const viewEl = document.getElementById('viewCounter'); const countdownEl = document.getElementById('countdownTimer'); const countdownSection = document.getElementById('countdownSection'); const NAMESPACE = "kartuuxx-profile-8f3d"; const KEY = "views"; const MILESTONE = 1000; // Change this to your target view count function updateCountdown() { fetch(`https://api.countapi.xyz/hit/${NAMESPACE}/${KEY}`) .then(res => res.json()) .then(data => { const currentViews = data.value; viewEl.textContent = `${currentViews.toLocaleString()} views`; // Calculate remaining views to milestone const remaining = MILESTONE - currentViews; // Always show the countdown section countdownSection.style.display = 'block'; if (remaining > 0) { countdownEl.textContent = `${remaining.toLocaleString()} views remaining`; countdownEl.style.color = 'var(--accent)'; } else if (remaining === 0) { countdownEl.textContent = '🎉 Milestone reached!'; countdownEl.style.color = '#4fd67a'; } else { countdownEl.textContent = `+${Math.abs(remaining).toLocaleString()} views`; countdownEl.style.color = '#4fd67a'; } }) .catch(() => { viewEl.textContent = '— views'; countdownSection.style.display = 'none'; }); } // Call immediately and on intervals updateCountdown(); setInterval(updateCountdown, 30000); // Refresh every 30 seconds })();
